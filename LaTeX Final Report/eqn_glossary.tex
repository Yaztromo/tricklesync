\chapter{Glossary of Algorithms, Equations, Symbols, and Variables}

The following is a glossary of algorithms, equations, symbols, and variables used in this thesis.  The first section where they are mentioned in is listed in brackets after each description, along with the equation number (where appropriate).

\begin{description}
\item[$\Delta t$] the absolute magnitude between two different tick intervals (\ref{when_to_sync})
\item[$\mu$] the mean value for a normal distribution (\ref{gaussian})
\item[$\sigma$] the variance for a normal distribution (\ref{gaussian})
\item[$\varphi_{\mu,\sigma^2}$] the normal distribution with mean $\mu$, and variance $\sigma$ (\ref{gaussian}, eqn. \ref{normal_dist})
\item[$C(r_i) < P(r_i)\cdot(V(r_{i_s})-V(r_{i_h}))\cdot k $] the inequality used to set the decision variable $x_i$.  When this inequality holds true, $x_i = 1$, otherwise $x_i = 0$ \ref{expected_sync_cost}, eqn. \ref{sync_cost_inequality})
\item[$C(R)$] a function which calculates the cost of transferring record $R$ on the currently selected network (\ref{selecting_records})
\item[$d$] a value which denotes a device, where $d \in \{s, h\}$ (\ref{selecting_records})
\item[$E_{hour}$] the expected number of events per hour for an event $E$ (\ref{bernoulli})
\item[$E_{max}$] the maximum number of permitted events per interval (\ref{poisson})
\item[$G(x)$] the ETS function evaluated to determine whether to synchronize during the current tick or not (\ref{when_to_sync}, eqn. \ref{threshold_function})
\item[$G(I(t)) > T$] the ETS threshold inequality, which compares the results of function $G$ to the threshold value $T$ (\ref{when_to_sync}, eqn. \ref{threshold_inequality})
\item[$I(t)$] a function which returns a fixed-sized interval of the day, which contains tick value $t$ (\ref{when_to_sync})
\item[$k$] the estimated average cost of using an out-of-date record on the mobile device (\ref{threshold_computation})
\item[$L_{entry}$] the time (in seconds since midnight) when the user enters locale $L$ (\ref{locales})
\item[$L_{expected accesses}$] the expected number of record accesses per hour when in local $L$ (\ref{locales})
\item[$L_{sync frequency}$] the frequency in which the user synchronizes their mobile device when in locale $L$ (\ref{locales})
\item[$\lim_{N_{cost_{current}}}G(x) = \infty$] the upper limit of $G(x)$ as $N_{cost_{current}}$ approaches zero (\ref{threshold_problems}, eqn. \ref{threshold_limit_eqn})
\item[$M$] the mobile device database corresponding to $S$ (\ref{databases})
\item[$M_k$] the record with an ID value of $k$, as stored in the mobile devices database (\ref{records})
\item[$N_{rate}$] the transfer rate of a network $N$, expressed in bytes-per-second (\ref{networks})
\item[$N_{cost}$] the cost of transferring data on a network $N$, expressed in dollars per byte (\ref{networks})
\item[$N_{cost_{current}}$] the cost of transmitting a single byte on the currently selected network (\ref{when_to_sync})
\item[$P_{access}$] a function which calculates the probability that a record will be accessed by the user (\ref{selecting_records})
\item[$P_{bernoulli}(1)=\frac{E_{hour}}{3600}$] the Bernoulli probability of an event happening during a single tick, based on the total expected events per hour (\ref{bernoulli}, eqn. \ref{bernoulli_prob})
\item[$P_{connection lost}(N)$] the probability that a loss of connection occurs in a one-second interval for network $N$ (\ref{networks})
\item[$P(n)>Z$] an inequality testing whether $n$ events have occurred during the current tick, based on the random variate $Z$ (\ref{poisson}, eqn. \ref{poissontrial})
\item[$P_{poisson}(n) = \sum_{k=0}^n \frac{(rt)^k}{k!}e^{-rt}$] the probability that $n$ events occur during interval $t$, with arrival rate $r$ (\ref{poisson}, eqn. \ref{poisson1})
\item[$Q$] a set of records on the mobile device which are out-of-date, and thus requiring synchronization (\ref{selecting_records})
\item[$|Q|$] the number of records in $Q$ (\ref{selecting_records})
\item[$r_i$] a record in set $Q$, requiring synchronization, with index value $i$ (\ref{selecting_records})
\item[$R_{ID}$] the unique identification number of record $R$ (\ref{records})
\item[$R_{size}$] the size, expressed in kilobytes, of record $R$ (\ref{records})
\item[$R_{revision}$] the modification revision of record $R$, expressed as a positive integer (\ref{records})
\item[$S$] the remote server database (\ref{databases})
\item[$S_k$] the record with an ID value of $k$, as stored in the remote server database (\ref{records})
\item[$S_{max arrivals}$] the maximum simultaneous record modifications permitted by the simulation in one tick in database $S$(\ref{databases})
\item[$S_{max}$] the largest record ID number in database $S$ (\ref{databases})
\item[$t$] a specific tick value during the day.
\item[$T$] a threshold value, as used by the ETS algorithm (\ref{when_to_sync})
\item[$T_{init}$] the initial threshold value for the ETS algorithm, equal to $8 \cdot 10^{14}$ (\ref{threshold_initialization}, eqn. \ref{threshold_init})
\item[$T_{lower}$] the lower value of the current ETS algorithm threshold range (\ref{threshold_initialization})
\item[$T_{lower_{init}} = \frac{T_{midpoint} - \displaystyle\frac{900}{max(N_{cost_{current}})}}{2}$] the threshold lower bound calculated after the first day of using the ETS algorithm (\ref{threshold_initialization}, eqn. \ref{threshold_init_lower})
\item[$T_{lower_{new}} = T_{lower}$] the new value for $T_{lower}$ when $T_{lower}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_decrease_lower})
\item[$T_{lower_{new}} = T_{lower} + \frac{T_{upper} - T_{lower}}{10}$] the new value for $T_{lower}$ when $T_{midpoint}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_diminish_lower})
\item[$T_{lower_{new}} = T_{midpoint}$] the new value for $T_{lower}$ when $T_{upper}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_increase_lower})
\item[$T_{midpoint}$] the mid-point of the current ETS algorithm threshold range (\ref{threshold_initialization})
\item[$T_{midpoint_{init}} = \frac{3600}{max(N_{cost_{current}})}$] the threshold midpoint calculated after the first day of using the ETS algorithm (\ref{threshold_initialization}, eqn. \ref{threshold_init_midpoint})
\item[$T_{midpoint_{new}} = T_{midpoint}$] the new value for $T_{midpoint}$ when $T_{midpoint}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_maintain_midpoint})
\item[$T_{midpoint_{new}} = \frac{T_{midpoint} - T_{lower}}{2}$] the new value for $T_{midpoint}$ when $T_{lower}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_decrease_midpoint})
\item[$T_{midpoint_{new}} = \frac{T_{upper} - T_{midpoint}}{2}$]  the new value for $T_{midpoint}$ when $T_{upper}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_increase_midpoint})
\item[$T_{upper}$] the upper value of the current ETS algorithm threshold range (\ref{threshold_initialization})
\item[$T_{upper_{init}} = \frac{\displaystyle\frac{43200}{max(N_{cost_{current}})} - T_{midpoint}}{2}$] the threshold upper bound calculated after the first day of using the ETS algorithm (\ref{threshold_initialization}, eqn. \item[$T_{upper_{new}} = T_{midpoint}$] the new value for $T_{upper}$ when $T_{lower}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_decrease_upper})
\item[$T_{upper_{new}} = T_{upper}$] the new value for $T_{upper}$ when $T_{upper}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_increase_upper})
\item[$T_{upper_{new}} = T_{upper} - \frac{T_{upper} - T_{lower}}{10}$] the new value for $T_{upper}$ when $T_{midpoint}$ is found to have the best cost reducing benefit (\ref{threshold_computation}, eqn. \ref{threshold_diminish_upper})
\item[$T_{upper} - T_{lower} > \epsilon$] an inequality used to determine if $T_{upper}$ and $T_{lower}$ are within a small bounds value $\epsilon$ of each other (\ref{threshold_computation}, eqn. \ref{upper_lower_within_epsilon})
\ref{threshold_init_upper})
\item[$U$] the user object for the current simulation run (\ref{user})
\item[$y=mx+b$] the standard linear equation, where $m$ is the slope of the line, and $b$ is the y-intersection; when used to describe the cost of synchronization, $m$ is the intangible cost of using out-of-date records, while $b$ is the cost of using network resources (\ref{cost_controller}, eqn. \ref{lineareq})
\item[$V(R, d)$] a function which calculates the version number of a record $R$ on device $d$ (\ref{selecting_records})
\item[$x_i$] a decision variable which decides whether record $r_i$ is to be synchronized, where $x_i \in \{0, 1\}$ (\ref{selecting_records})
\item[$XC = \sum_{i=1}^n C(r_i)x_i+P_{access}(r_i)\overline{x_i}\cdot(V(r_{i}, s)-V(r_{i}, h))\cdot k$] the expected cost of synchronizing the set $Q$ at a specific tick (\ref{selecting_records}, eqn. \ref{expected_sync_cost})
\item[$Z$] a uniformly distributed random number, as selected from a PRNG (\ref{poisson})
\end{description}
